      def create
        title = MenuItem.new('Options')
        title.submenu = menu
        add_item('Collapsed nodes', nil, CheckMenuItem, &method(:collapsed_nodes))
        @pretty_item = add_item('Pretty saving', nil, CheckMenuItem,
